# REPL v2 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the basic `input_line` REPL in `bin/main.ml` with a full TUI REPL using notty — two-pane layout with scope panel, syntax highlighting, readline-style editing, history, magic `v` result variables, and a non-TUI fallback for piped/CI contexts.

**Architecture:** New `lib/repl/` OCaml library containing 8 focused modules. `bin/main.ml` delegates to `March_repl.Repl.run` when invoked with no file arguments. notty owns the terminal exclusively (no linenoise — they conflict for raw mode ownership). Line editing is a pure state machine in `input.ml`.

**Tech Stack:** OCaml 5.3.0, notty (>= 0.2.3) + notty.unix, alcotest (existing), dune 3.0.

**Spec:** `specs/repl_design.md`

```ocaml
(** Multi-line input completion heuristics for the REPL.
    Migrated from bin/main.ml. *)

(** Count exact whole-word occurrences of [tok] in [buf]. *)
let count_token tok buf =
  let words = Str.split (Str.regexp "[^a-zA-Z0-9_']") buf in
  List.length (List.filter (( = ) tok) words)

(** Count lines whose last word is "with" — match expression openers. *)
let count_match_with buf =
  let lines = String.split_on_char '\n' buf in
  List.length (List.filter (fun line ->
      let words = Str.split (Str.regexp "[^a-zA-Z0-9_']") (String.trim line) in
      match List.rev words with
      | "with" :: _ -> true
      | _           -> false
    ) lines)

(** Net depth of open blocks: do/with openers minus end closers.
    Positive means we are inside an unclosed block.
    Known limitation: tokens inside string literals are miscounted. *)
let do_end_depth buf =
  count_token "do" buf + count_match_with buf - count_token "end" buf

(** Last non-blank line in [buf], trimmed. *)
let last_non_blank_line buf =
  let lines = String.split_on_char '\n' buf in
  match List.rev (List.filter (fun l -> String.trim l <> "") lines) with
  | []     -> ""
  | l :: _ -> String.trim l

(** True if the last non-blank line ends with the token "with". *)
let ends_with_with buf =
  let l = last_non_blank_line buf in
  let words = String.split_on_char ' ' (String.trim l) in
  match List.rev words with
  | "with" :: _ -> true
  | _            -> false

(** True if the last non-blank line starts with '|' (match arm continuation). *)
let starts_with_pipe buf =
  let l = last_non_blank_line buf in
  String.length l > 0 && l.[0] = '|'

(** True when [buf] appears syntactically complete (safe to submit). *)
let is_complete buf =
  do_end_depth buf <= 0
  && not (ends_with_with buf)
  && not (starts_with_pipe buf)
```

- [ ] **Step 3: Implement lib/repl/result_vars.ml**

We use string representations of values and types for now (the REPL stores them for display; the evaluator will look up the actual values separately via the `env` ref).

```ocaml
(** Magic result variable history: v, v(1), v(2), ...
    Stores string representations of values and their type strings.
    The actual eval values are stored separately in the eval env under the name "v";
    this module tracks the display strings and type strings for the type checker
    and scope panel. *)

let max_history = 100

type t = {
  mutable entries : (string * string) array;  (* (value_str, type_str) *)
  mutable size    : int;
  mutable head    : int;  (* index of most recent entry *)
}

let create () = { entries = Array.make max_history ("", ""); size = 0; head = 0 }

let push_string h value_str type_str =
  let next = (h.head + 1) mod max_history in
  h.entries.(next) <- (value_str, type_str);
  h.head <- next;
  if h.size < max_history then h.size <- h.size + 1

let get h n =
  if n < 0 || n >= h.size then None
  else
    let idx = ((h.head - n) + max_history) mod max_history in
    let (v, ty) = h.entries.(idx) in
    Some (v, ty)

let length h = h.size
```

Create `lib/repl/result_vars.mli`:

```ocaml
val push_string : t -> string -> string -> unit

val get : t -> int -> (string * string) option
(** [get h n] returns [(value_str, type_str)] for the nth-previous result
    (0 = most recent). Returns [None] if out of bounds. *)
val length : t -> int
```

This substitution also happens before typechecking (substituting with typed vars from the typecheck env).

- [ ] **Step 6: Parse `v` and `v(N)` in parser.mly (REPL mode only)**

```menhir
      else (* normal function call *) EApp (...) }
| LOWER_IDENT
    { if $1 = "v" then EResultRef None
      else EVar (spanify $1) }
```

However, modifying the grammar this way for REPL-only behavior is messy. The cleanest approach: treat `v` and `v(N)` as a post-parse rewrite in `repl.ml`. After parsing, walk the AST and replace `EVar { txt = "v" }` with `EResultRef None` and `EApp (EVar { txt = "v" }, [EInt n])` with `EResultRef (Some n)`.

Implement this rewrite in a new function `repl.ml`:
```ocaml
(** Rewrite v and v(N) references into EResultRef nodes.
    This runs after parsing, before desugar. *)
let rec rewrite_result_refs expr =
  match expr with
  | EVar n when n.txt = "v" -> EResultRef None
  | EApp ({ expr = EVar n; _ }, [{ expr = EInt idx; _ }]) when n.txt = "v" ->
    EResultRef (Some idx)
  | _ -> map_expr rewrite_result_refs expr
```

```ocaml
(** Input history: ring buffer with Up/Down navigation and NUL-separated file persistence. *)

type t = {
  mutable entries  : string array;  (* ring buffer, NUL = empty slot *)
  max_size         : int;
  mutable count    : int;           (* number of valid entries *)
  mutable write_at : int;           (* next write position *)
  mutable nav_pos  : int;           (* current navigation position, -1 = at bottom *)
}

let create ~max_size =
  { entries = Array.make max_size ""; max_size; count = 0; write_at = 0; nav_pos = -1 }

(** Add an entry. Ignores blank lines and duplicates of the most recent entry. *)
let add h entry =
  if String.trim entry = "" then ()
  else begin
    let most_recent =
      if h.count = 0 then None
      else
        let idx = ((h.write_at - 1) + h.max_size) mod h.max_size in
        Some h.entries.(idx)
    in
    if most_recent = Some entry then ()
    else begin
      h.entries.(h.write_at) <- entry;
      h.write_at <- (h.write_at + 1) mod h.max_size;
      if h.count < h.max_size then h.count <- h.count + 1;
      h.nav_pos <- -1
    end
  end

(** Navigate one step older. Returns None if at the oldest entry. *)
let prev h =
  if h.count = 0 then None
  else
    let next_pos =
      if h.nav_pos = -1 then 0
      else h.nav_pos + 1
    in
    if next_pos >= h.count then None
    else begin
      h.nav_pos <- next_pos;
      let idx = ((h.write_at - 1 - next_pos) + h.max_size * 2) mod h.max_size in
      Some h.entries.(idx)
    end

(** Navigate one step newer. Returns None if already at the newest. *)
let next h =
  if h.nav_pos <= 0 then begin
    h.nav_pos <- -1;
    None
  end else begin
    h.nav_pos <- h.nav_pos - 1;
    let idx = ((h.write_at - 1 - h.nav_pos) + h.max_size * 2) mod h.max_size in
    Some h.entries.(idx)
  end

(** Reset navigation position to bottom (editing new input). *)
let reset_pos h = h.nav_pos <- -1

(** Save history to file. Entries separated by NUL bytes, oldest first. *)
let save h path =
  let entries =
    let acc = ref [] in
    for i = 0 to h.count - 1 do
      let idx = ((h.write_at - 1 - i) + h.max_size * 2) mod h.max_size in
      acc := h.entries.(idx) :: !acc
    done;
    !acc  (* oldest first *)
  in
  let content = String.concat "\x00" entries in
  let oc = open_out path in
  output_string oc content;
  close_out oc

(** Load history from file (NUL-separated). Clears existing entries first. *)
let load h path =
  match (try Some (
    let ic = open_in path in
    let n = in_channel_length ic in
    let buf = Bytes.create n in
    really_input ic buf 0 n;
    close_in ic;
    Bytes.to_string buf
  ) with _ -> None) with
  | None -> ()
  | Some content ->
    let parts = String.split_on_char '\x00' content in
    List.iter (fun entry ->
      if entry <> "" then add h entry
    ) parts
```

```ocaml
(** Syntax highlighting: tokenize input with March_lexer and map tokens to
    Notty attributes. Best-effort — falls back to default color on lex error. *)

open Notty
open Notty.A

(* Color scheme *)
let attr_keyword   = st bold ++ fg magenta
let attr_type      = fg cyan
let attr_ctor      = st bold ++ fg cyan
let attr_string    = fg green
let attr_number    = fg yellow
let attr_atom      = fg blue
let attr_comment   = st dim
let attr_operator  = st bold
let attr_default   = empty

(** Classify a token into an attribute. *)
let attr_of_token tok =
  let open March_parser.Parser in
  match tok with
  | FN | LET | DO | END | IF | THEN | ELSE | MATCH | WITH | WHEN
  | TYPE | MOD | ACTOR | ON | SEND | SPAWN | STATE | INIT | RESPOND
  | PROTOCOL | LOOP | LINEAR | AFFINE | PUB | INTERFACE | IMPL | SIG
  | EXTERN | UNSAFE | AS | USE -> attr_keyword
  | INT _ | FLOAT _ -> attr_number
  | STRING _ | INTERP_START _ | INTERP_MID _ | INTERP_END _ -> attr_string
  | ATOM _ -> attr_atom
  | BOOL _ -> attr_keyword
  | UPPER_IDENT _ -> attr_type
  | _ -> attr_default

(** Tokenize [src] and produce a styled notty image.
    On lex error, the successfully tokenized prefix is styled and the rest
    is rendered in default color. *)
let highlight src =
  if src = "" then I.empty
  else
    let lexbuf = Lexing.from_string src in
    let buf = Buffer.create (String.length src) in
    let images = ref [] in
    let flush_buf attr =
      let s = Buffer.contents buf in
      if s <> "" then begin
        images := I.string attr s :: !images;
        Buffer.clear buf
      end
    in
    (* Walk through the source character by character, using lexeme positions
       to extract token text *)
    let pos = ref 0 in
    let src_len = String.length src in
    let done_ = ref false in
    while not !done_ do
      if !pos >= src_len then done_ := true
      else
        (match (try
           let tok = March_lexer.Lexer.token lexbuf in
           let start = lexbuf.Lexing.lex_start_p.Lexing.pos_cnum in
           let stop  = lexbuf.Lexing.lex_curr_p.Lexing.pos_cnum in
           Some (tok, start, stop)
         with _ -> None)
        with
        | None ->
          (* Lex error — emit remainder as default *)
          flush_buf attr_default;
          let rest = String.sub src !pos (src_len - !pos) in
          images := I.string attr_default rest :: !images;
          done_ := true
        | Some (March_parser.Parser.EOF, _, _) -> done_ := true
        | Some (tok, start, stop) ->
          (* Emit any gap between pos and start as default (whitespace/newlines) *)
          if start > !pos then begin
            let gap = String.sub src !pos (start - !pos) in
            images := I.string attr_default gap :: !images
          end;
          let lexeme = String.sub src start (stop - start) in
          let attr = attr_of_token tok in
          images := I.string attr lexeme :: !images;
          pos := stop)
    done;
    List.fold_left I.(<|>) I.empty (List.rev !images)
```

```ocaml
(** Readline-style input state machine over notty key events.
    Pure function: takes state + key → new state + action.
    No terminal I/O. *)

type input_action =
  | Noop
  | Redraw
  | Submit of string
  | EOF
  | Complete
  | HistoryPrev
  | HistoryNext
  | HistorySearch

type state = {
  buffer      : string;
  cursor      : int;
  kill_ring   : string list;
  history_pos : int;
  multiline_buf : string list;
}

let empty = { buffer = ""; cursor = 0; kill_ring = []; history_pos = -1; multiline_buf = [] }

let full_buffer s =
  let lines = List.rev (s.buffer :: s.multiline_buf) in
  String.concat "\n" lines

(** Insert string at cursor position. *)
let insert_at s str =
  let b = s.buffer in
  let c = s.cursor in
  let n = String.length str in
  { s with
    buffer = String.sub b 0 c ^ str ^ String.sub b c (String.length b - c);
    cursor = c + n }

(** Delete char before cursor. *)
let delete_back s =
  if s.cursor = 0 then s
  else
    { s with
      buffer = String.sub s.buffer 0 (s.cursor - 1)
               ^ String.sub s.buffer s.cursor (String.length s.buffer - s.cursor);
      cursor = s.cursor - 1 }

(** Delete char at cursor. *)
let delete_forward s =
  let len = String.length s.buffer in
  if s.cursor >= len then s
  else
    { s with buffer =
        String.sub s.buffer 0 s.cursor
        ^ String.sub s.buffer (s.cursor + 1) (len - s.cursor - 1) }

(** Find the start of the word to the left of cursor. *)
let word_start_left s =
  let b = s.buffer and c = s.cursor in
  let i = ref (c - 1) in
  while !i > 0 && b.[!i - 1] = ' ' do decr i done;
  while !i > 0 && b.[!i - 1] <> ' ' do decr i done;
  !i

(** Find the end of the word to the right of cursor. *)
let word_end_right s =
  let b = s.buffer and c = s.cursor in
  let len = String.length b in
  let i = ref c in
  while !i < len && b.[!i] = ' ' do incr i done;
  while !i < len && b.[!i] <> ' ' do incr i done;
  !i

(** Push to kill ring (max 8). *)
let push_kill s killed =
  let ring = killed :: (if List.length s.kill_ring >= 8
                        then List.filteri (fun i _ -> i < 7) s.kill_ring
                        else s.kill_ring) in
  { s with kill_ring = ring }

let handle_key s key =
  match key with
  (* Printable character *)
  | `Uchar uc ->
    let str = let b = Buffer.create 4 in Buffer.add_utf_8_uchar b uc; Buffer.contents b in
    (insert_at s str, Redraw)

  (* Cursor movement *)
  | `Arrow `Left  -> ({ s with cursor = max 0 (s.cursor - 1) }, Redraw)
  | `Arrow `Right -> ({ s with cursor = min (String.length s.buffer) (s.cursor + 1) }, Redraw)
  | `Arrow `Up    -> (s, HistoryPrev)
  | `Arrow `Down  -> (s, HistoryNext)

  | `Home | `Ctrl `A -> ({ s with cursor = 0 }, Redraw)
  | `End  | `Ctrl `E -> ({ s with cursor = String.length s.buffer }, Redraw)

  (* Word movement *)
  | `Meta `Left  | `Ctrl `B ->  (* Alt+Left = word back *)
    ({ s with cursor = word_start_left s }, Redraw)
  | `Meta `Right | `Ctrl `F ->  (* Alt+Right = word forward *)
    ({ s with cursor = word_end_right s }, Redraw)

  (* Deletion *)
  | `Backspace -> (delete_back s, Redraw)
  | `Delete    -> (delete_forward s, Redraw)

  | `Ctrl `W ->  (* Kill word backward *)
    let start = word_start_left s in
    let killed = String.sub s.buffer start (s.cursor - start) in
    let s' = push_kill s killed in
    let s' = { s' with
      buffer = String.sub s.buffer 0 start
               ^ String.sub s.buffer s.cursor (String.length s.buffer - s.cursor);
      cursor = start } in
    (s', Redraw)

  | `Ctrl `K ->  (* Kill to end of line *)
    let killed = String.sub s.buffer s.cursor (String.length s.buffer - s.cursor) in
    let s' = push_kill s killed in
    let s' = { s' with buffer = String.sub s.buffer 0 s.cursor } in
    (s', Redraw)

  | `Ctrl `U ->  (* Kill whole line *)
    let killed = s.buffer in
    let s' = push_kill s killed in
    ({ s' with buffer = ""; cursor = 0 }, Redraw)

  | `Ctrl `Y ->  (* Yank most recent kill *)
    (match s.kill_ring with
     | []     -> (s, Noop)
     | k :: _ -> (insert_at s k, Redraw))

  (* Submit *)
  | `Enter ->
    let full = full_buffer s in
    if March_repl.Multiline.is_complete full then
      (empty, Submit full)
    else
      (* Add current line to multiline buffer, start fresh *)
      let s' = { s with
        buffer = "";
        cursor = 0;
        multiline_buf = s.buffer :: s.multiline_buf } in
      (s', Redraw)

  (* EOF *)
  | `Ctrl `D ->
    if s.buffer = "" && s.multiline_buf = []
    then (s, EOF)
    else (delete_forward s, Redraw)

  (* Completion *)
  | `Tab -> (s, Complete)

  (* History search *)
  | `Ctrl `R -> (s, HistorySearch)

  (* Clear/redraw *)
  | `Ctrl `L -> (s, Redraw)

  (* Ignore everything else *)
  | _ -> (s, Noop)
```

```ocaml
(** Notty-based two-pane TUI: left pane = REPL transcript, right pane = scope. *)

open Notty
open Notty.A

type t = {
  term    : Notty_unix.Term.t;
  mutable w : int;
  mutable h : int;
}

type pane_content = {
  history       : I.t list;
  input_img     : I.t;
  scope         : (string * string * string) list;
  result_latest : (string * string) option;
  prompt        : string;
  status        : string;
}

let create () =
  let term = Notty_unix.Term.create () in
  let (w, h) = Notty_unix.Term.size term in
  { term; w; h }

let close tui = Notty_unix.Term.release tui.term

let width  tui = tui.w
let height tui = tui.h

(** Render a string with attributes as a notty image, right-padded to [w] cols. *)
let padded_string attr s w =
  let img = I.string attr s in
  let pad = max 0 (w - I.width img) in
  I.(img <|> char empty ' ' pad 1)

(** Build the left (transcript) pane. *)
let left_pane content pane_w pane_h =
  let prompt_img = I.string A.(st bold ++ fg blue) content.prompt in
  let input_line = I.(prompt_img <|> content.input_img) in
  let all_lines  = content.history @ [input_line] in
  (* Take the last pane_h lines *)
  let visible = let n = List.length all_lines in
    if n <= pane_h then all_lines
    else let skip = n - pane_h in
      List.filteri (fun i _ -> i >= skip) all_lines
  in
  (* Pad each line to pane_w, then vcat *)
  let padded = List.map (fun img ->
    let pad = max 0 (pane_w - I.width img) in
    I.(img <|> char empty ' ' pad 1)
  ) visible in
  (* Pad to full pane height *)
  let filler_rows = max 0 (pane_h - List.length padded) in
  let filler = I.char empty ' ' pane_w filler_rows in
  let content_img = List.fold_left I.(<->) I.empty padded in
  I.(filler <-> content_img)

(** Build the right (scope) pane. *)
let right_pane content pane_w pane_h =
  let header = padded_string A.(st bold) "Variables in Scope" pane_w in
  let sep    = I.string A.dim (String.make pane_w '─') in
  let v_row  = match content.result_latest with
    | None -> []
    | Some (ty, vs) ->
      let s = Printf.sprintf "v : %s = %s" ty vs in
      [padded_string A.(fg green) s pane_w]
  in
  let scope_rows = List.map (fun (name, ty, vs) ->
    let s = Printf.sprintf "%s : %s = %s" name ty vs in
    [padded_string A.empty s pane_w]
  ) content.scope |> List.flatten in
  let rows = [header; sep] @ v_row @ scope_rows in
  let visible = if List.length rows <= pane_h then rows
    else List.filteri (fun i _ -> i < pane_h) rows in
  let filler_rows = max 0 (pane_h - List.length visible) in
  let filler = I.char empty ' ' pane_w filler_rows in
  List.fold_left I.(<->) I.empty (visible @ [filler])

let render tui content =
  let w = tui.w and h = tui.h in
  let status_h = 1 in
  let pane_h   = h - status_h in
  let right_w  = w / 3 in
  let sep_w    = 1 in
  let left_w   = w - right_w - sep_w in

  let left  = left_pane  content left_w  pane_h in
  let sep   = I.char A.dim '│' sep_w pane_h in
  let right = right_pane content right_w pane_h in

  let panes = I.(left <|> sep <|> right) in

  let status_img =
    padded_string A.dim
      (Printf.sprintf " %s" content.status) w
  in

  let screen = I.(panes <-> status_img) in
  Notty_unix.Term.image tui.term screen

let next_event tui =
  match Notty_unix.Term.event tui.term with
  | `Key (key, _mods) -> `Key key
  | `Resize (w, h) ->
    tui.w <- w; tui.h <- h;
    `Resize (w, h)
  | `Mouse _ -> next_event tui  (* ignore mouse for now *)
  | `Paste _  -> next_event tui
  | `End      -> `End
```

```ocaml
(** Tab completion engine. Context-aware, priority-ordered. *)

let repl_commands = [":quit"; ":q"; ":env"; ":help"; ":type"; ":clear"; ":reset"; ":load"; ":save"]

let keywords = [
  "fn"; "do"; "end"; "let"; "match"; "with"; "if"; "then"; "else";
  "mod"; "actor"; "type"; "pub"; "use"; "impl"; "interface"; "sig";
  "spawn"; "send"; "on"; "state"; "init"; "respond"; "when"; "as"; "use"
]

let starts_with prefix s =
  let pl = String.length prefix and sl = String.length s in
  pl <= sl && String.sub s 0 pl = prefix

(** [complete prefix scope] returns completions for [prefix].
    [scope] is a list of (name, type_str) pairs from the current env. *)
let complete prefix scope =
  if String.length prefix > 0 && prefix.[0] = ':' then
    (* REPL command completion *)
    List.filter (starts_with prefix) repl_commands
  else
    let kw_matches = List.filter (starts_with prefix) keywords in
    let scope_matches = List.filter_map (fun (name, _ty) ->
      if starts_with prefix name then Some name else None
    ) scope in
    (* Deduplicate, scope before keywords for priority *)
    let seen = Hashtbl.create 16 in
    List.filter (fun s ->
      if Hashtbl.mem seen s then false
      else (Hashtbl.add seen s (); true)
    ) (scope_matches @ kw_matches)
```
